/**@module model/fridge*/
import mongoose from "mongoose";
const Schema = mongoose.Schema;

/**
 * Schema for Fridge object. May also be called Pantry object.
 * @author Tyler Del Rosario
 * route_id 11 character string that must be generated by {@link RouteID}
 */
const FridgeSchema = new Schema({
    route_id: {type: String, required: true, unique: true},
    owner_id: {type: Schema.Types.ObjectId, ref: "User", required: true},
    ingredients: [{type: Schema.Types.ObjectId, ref: "Ingredient"}]
});

FridgeSchema.virtual("url").get(function() {
    return `/fridge/${this._id}`;
});

export default mongoose.model("Fridge", FridgeSchema);
